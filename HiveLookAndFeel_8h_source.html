72 // ---- Buttons ---------------------------------------------------------
73 void drawButtonBackground (juce::Graphics& g, juce::Button& button,
74 const juce::Colour& backgroundColour,
75 bool isHighlighted, bool isDown) override
76 {
77 auto r = button.getLocalBounds().toFloat().reduced (1.0f);
78 const float corner = 7.0f;
79
80 const bool isAccent = button.getProperties().getWithDefault ("accent", false);
81
82 if (isAccent)
83 {
84 // Soft drop shadow
85 juce::Path shadowPath;
86 shadowPath.addRoundedRectangle (r.translated (0.0f, 1.5f), corner);
87 juce::DropShadow ds (HarmoniaPalette::accent.withAlpha (0.30f), 10, { 0, 2 });
88 ds.drawForPath (g, shadowPath);
89
90 // Cyan gradient fill
91 juce::ColourGradient grad (HarmoniaPalette::accentHover, r.getX(), r.getY(),
92 HarmoniaPalette::accentSecond, r.getRight(), r.getBottom(), false);
93 g.setGradientFill (grad);
94 g.fillRoundedRectangle (r, corner);
95
96 // Rim light en haut (apple-like)
97 g.setColour (juce::Colours::white.withAlpha (0.18f));
98 g.drawLine (r.getX() + corner, r.getY() + 1.0f,
99 r.getRight() - corner, r.getY() + 1.0f, 1.0f);
100
101 if (isHighlighted)
102 {
103 g.setColour (HarmoniaPalette::accent.withAlpha (0.40f));
104 g.drawRoundedRectangle (r.expanded (1.5f), corner + 1.5f, 2.0f);
105 }
106 }
107 else
108 {
109 // Subtle gradient pour profondeur
110 juce::Colour top = backgroundColour.brighter (0.05f);
111 juce::Colour bot = backgroundColour.darker (0.10f);
112 if (isDown) { top = backgroundColour.darker (0.10f); bot = backgroundColour.darker (0.20f); }
113 else if (isHighlighted) { top = backgroundColour.brighter (0.15f); bot = backgroundColour.brighter (0.05f); }
114
115 juce::ColourGradient grad (top, r.getX(), r.getY(), bot, r.getX(), r.getBottom(), false);
116 g.setGradientFill (grad);
117 g.fillRoundedRectangle (r, corner);
118
119 // Rim light
120 g.setColour (juce::Colours::white.withAlpha (0.06f));
121 g.drawLine (r.getX() + corner, r.getY() + 1.0f,
122 r.getRight() - corner, r.getY() + 1.0f, 1.0f);
123
124 g.setColour (isHighlighted ? HarmoniaPalette::accent.withAlpha (0.6f) : HarmoniaPalette::border);
125 g.drawRoundedRectangle (r, corner, 1.0f);
126 }
127 }

129 // ---- Knob ------------------------------------------------------------
130 // Hive style + specular highlight Apple : corps métallique, arc cyan
131 // (ou couleur custom via property "tint"), drop shadow externe diffuse,
132 // petit specular en haut-gauche du body.
133 void drawRotarySlider (juce::Graphics& g, int x, int y, int width, int height,
134 float sliderPos, float startAngle, float endAngle,
135 juce::Slider& slider) override
136 {
137 auto bounds = juce::Rectangle<int> (x, y, width, height).toFloat().reduced (3.0f);
138 const float radius = juce::jmin (bounds.getWidth(), bounds.getHeight()) * 0.5f;
139 const auto centre = bounds.getCentre();
140 const float arcThick = juce::jmax (3.0f, radius * 0.10f);
141 const float arcRadius = radius - arcThick * 0.5f - 1.0f;
142 const float bodyRadius = arcRadius - arcThick - 2.0f;
143
144 const bool bipolar = slider.getProperties().getWithDefault ("bipolar", false);
145 const bool hover = slider.isMouseOverOrDragging();
146 const auto tintProp = slider.getProperties().getWithDefault ("tint", juce::var());
147 const auto tint = tintProp.isVoid()
148 ? HarmoniaPalette::accent
149 : juce::Colour::fromString (tintProp.toString());
150
151 // 1) Drop shadow diffuse
152 {
153 juce::Path shadowPath;
154 shadowPath.addEllipse (juce::Rectangle<float> (bodyRadius * 2.0f, bodyRadius * 2.0f)
155 .withCentre (centre.translated (0.0f, 2.0f)));
156 juce::DropShadow ds (juce::Colours::black.withAlpha (0.55f), 10, { 0, 3 });
157 ds.drawForPath (g, shadowPath);
158 }
159
160 // 2) Track sombre
161 {
162 juce::Path track;
163 track.addCentredArc (centre.x, centre.y, arcRadius, arcRadius,
164 0.0f, startAngle, endAngle, true);
165 g.setColour (HarmoniaPalette::knobTrack);
166 g.strokePath (track, juce::PathStrokeType (arcThick,
167 juce::PathStrokeType::curved,
168 juce::PathStrokeType::butt));
169 }
170
171 // 3) Corps métallique mat
172 {
173 juce::ColourGradient bodyGrad (HarmoniaPalette::knobBgHi.brighter (0.06f),
174 centre.translated (-bodyRadius * 0.30f, -bodyRadius * 0.40f),
175 HarmoniaPalette::knobBg.darker (0.30f),
176 centre.translated (bodyRadius * 0.40f, bodyRadius * 0.50f), true);
177 g.setGradientFill (bodyGrad);
178 g.fillEllipse (juce::Rectangle<float> (bodyRadius * 2.0f, bodyRadius * 2.0f).withCentre (centre));
179
180 // Specular highlight (small white arc top-left, "Apple modern" feel)
181 juce::Path spec;
182 spec.addCentredArc (centre.x, centre.y,
183 bodyRadius * 0.92f, bodyRadius * 0.92f,
184 0.0f,
185 juce::MathConstants<float>::pi * 1.20f,
186 juce::MathConstants<float>::pi * 1.65f, true);
187 g.setColour (juce::Colours::white.withAlpha (0.10f));
188 g.strokePath (spec, juce::PathStrokeType (1.2f, juce::PathStrokeType::curved,
189 juce::PathStrokeType::rounded));
190
191 g.setColour (HarmoniaPalette::border);
192 g.drawEllipse (juce::Rectangle<float> (bodyRadius * 2.0f, bodyRadius * 2.0f).withCentre (centre), 1.0f);
193 }
194
195 // 4) Active arc (tint, bipolaire = warm côté gauche)
196 const float currentAngle = startAngle + sliderPos * (endAngle - startAngle);
197 juce::Path arc;
198 const float midAngle = (startAngle + endAngle) * 0.5f;
199 bool onLeftSide = false;
200
201 if (bipolar)
202 {
203 const float fromA = std::min (midAngle, currentAngle);
204 const float toA = std::max (midAngle, currentAngle);
205 arc.addCentredArc (centre.x, centre.y, arcRadius, arcRadius, 0.0f, fromA, toA, true);
206 onLeftSide = currentAngle < midAngle;
207 }
208 else
209 {
210 arc.addCentredArc (centre.x, centre.y, arcRadius, arcRadius, 0.0f, startAngle, currentAngle, true);
211 }
212
213 const auto arcCol = onLeftSide ? HarmoniaPalette::warm : tint;
214 g.setColour (arcCol);
215 g.strokePath (arc, juce::PathStrokeType (arcThick,
216 juce::PathStrokeType::curved,
217 juce::PathStrokeType::butt));
218
219 // Glow toujours visible
220 const float glowAlpha = hover ? 0.32f : 0.16f;
221 g.setColour (arcCol.withAlpha (glowAlpha));
222 g.strokePath (arc, juce::PathStrokeType (arcThick + 5.0f,
223 juce::PathStrokeType::curved,
224 juce::PathStrokeType::butt));
225
226 // 5) Indicateur (ligne fine + cap lumineux)
227 {
228 const float innerR = bodyRadius * 0.30f;
229 const float outerR = bodyRadius * 0.80f;
230 const float thick = juce::jmax (1.5f, radius * 0.05f);
231
232 const float cosA = std::cos (currentAngle - juce::MathConstants<float>::halfPi);
233 const float sinA = std::sin (currentAngle - juce::MathConstants<float>::halfPi);
234 const juce::Point<float> p1 (centre.x + cosA * innerR, centre.y + sinA * innerR);
235 const juce::Point<float> p2 (centre.x + cosA * outerR, centre.y + sinA * outerR);
236
237 g.setColour (HarmoniaPalette::textPrimary);
238 g.drawLine (p1.x, p1.y, p2.x, p2.y, thick);
239
240 const float capR = thick * 1.2f;
241 g.setColour (arcCol.withAlpha (0.55f));
242 g.fillEllipse (p2.x - capR * 1.6f, p2.y - capR * 1.6f, capR * 3.2f, capR * 3.2f);
243 g.setColour (HarmoniaPalette::textPrimary);
244 g.fillEllipse (p2.x - capR, p2.y - capR, capR * 2.0f, capR * 2.0f);
245 }
246 }

249 void drawLinearSlider (juce::Graphics& g, int x, int y, int width, int height,
250 float sliderPos, float minSliderPos, float maxSliderPos,
251 const juce::Slider::SliderStyle style, juce::Slider& slider) override
252 {
253 juce::ignoreUnused (minSliderPos, maxSliderPos, slider);
254
255 if (style == juce::Slider::LinearVertical)
256 {
257 const float trackW = 6.0f;
258 const float cx = (float) x + (float) width * 0.5f;
259 auto track = juce::Rectangle<float> (cx - trackW * 0.5f, (float) y + 4.0f,
260 trackW, (float) height - 8.0f);
261
262 // Drop shadow inset (effet "creusé")
263 g.setColour (juce::Colours::black.withAlpha (0.45f));
264 g.fillRoundedRectangle (track.translated (0.0f, 1.0f), trackW * 0.5f);
265
266 g.setColour (HarmoniaPalette::knobTrack);
267 g.fillRoundedRectangle (track, trackW * 0.5f);
268 g.setColour (HarmoniaPalette::border);
269 g.drawRoundedRectangle (track, trackW * 0.5f, 1.0f);
270
271 const float thumbY = juce::jlimit (track.getY(), track.getBottom(), sliderPos);
272 auto fill = juce::Rectangle<float> (track.getX(), thumbY,
273 track.getWidth(),
274 track.getBottom() - thumbY);
275
276 // Use knob's tint if set, else cyan
277 const auto tintProp = slider.getProperties().getWithDefault ("tint", juce::var());
278 const auto tint = tintProp.isVoid()
279 ? HarmoniaPalette::accent
280 : juce::Colour::fromString (tintProp.toString());
281
282 juce::ColourGradient fillGrad (tint, fill.getX(), fill.getY(),
283 tint.darker (0.10f), fill.getX(), fill.getBottom(), false);
284 g.setGradientFill (fillGrad);
285 g.fillRoundedRectangle (fill, trackW * 0.5f);
286
287 g.setColour (tint.withAlpha (0.20f));
288 g.fillRoundedRectangle (fill.expanded (3.0f, 0.0f), trackW * 0.5f + 3.0f);
289
290 // Thumb : pill rectangulaire avec gradient + rim light + grip lines
291 const float thumbW = 24.0f;
292 const float thumbH = 10.0f;
293 auto thumb = juce::Rectangle<float> (cx - thumbW * 0.5f, thumbY - thumbH * 0.5f,
294 thumbW, thumbH);
295
296 // Drop shadow
297 juce::Path thumbPath;
298 thumbPath.addRoundedRectangle (thumb.translated (0.0f, 1.0f), 3.0f);
299 juce::DropShadow ds (juce::Colours::black.withAlpha (0.60f), 5, { 0, 1 });
300 ds.drawForPath (g, thumbPath);
301
302 juce::ColourGradient thumbGrad (HarmoniaPalette::textPrimary.brighter (0.10f),
303 thumb.getX(), thumb.getY(),
304 HarmoniaPalette::textSecondary.darker (0.20f),
305 thumb.getX(), thumb.getBottom(), false);
306 g.setGradientFill (thumbGrad);
307 g.fillRoundedRectangle (thumb, 3.0f);
308
309 // Rim light
310 g.setColour (juce::Colours::white.withAlpha (0.40f));
311 g.drawLine (thumb.getX() + 3.0f, thumb.getY() + 1.0f,
312 thumb.getRight() - 3.0f, thumb.getY() + 1.0f, 1.0f);
313
314 g.setColour (HarmoniaPalette::border);
315 g.drawRoundedRectangle (thumb, 3.0f, 1.0f);
316
317 // Grip lines
318 g.setColour (HarmoniaPalette::knobTrack.withAlpha (0.7f));
319 for (int i = -1; i <= 1; ++i)
320 g.drawHorizontalLine ((int) thumb.getCentreY() + i * 2,
321 thumb.getX() + 5.0f, thumb.getRight() - 5.0f);
322
323 return;
324 }
325
326 if (style == juce::Slider::LinearHorizontal || style == juce::Slider::LinearBar)
327 {
328 auto track = juce::Rectangle<float> ((float) x, (float) y + height * 0.5f - 2.0f,
329 (float) width, 4.0f);
330 g.setColour (HarmoniaPalette::knobTrack);
331 g.fillRoundedRectangle (track, 2.0f);
332
333 auto fill = track.withWidth (juce::jmax (4.0f, sliderPos - (float) x));
334 g.setColour (HarmoniaPalette::accent);
335 g.fillRoundedRectangle (fill, 2.0f);
336
337 const float thumbR = 6.0f;
338 g.setColour (HarmoniaPalette::accent.withAlpha (0.25f));
339 g.fillEllipse (sliderPos - thumbR - 2.0f, (float) y + height * 0.5f - thumbR - 2.0f,
340 (thumbR + 2.0f) * 2.0f, (thumbR + 2.0f) * 2.0f);
341 g.setColour (HarmoniaPalette::textPrimary);
342 g.fillEllipse (sliderPos - thumbR, (float) y + height * 0.5f - thumbR, thumbR * 2.0f, thumbR * 2.0f);
343 return;
344 }
345
346 LookAndFeel_V4::drawLinearSlider (g, x, y, width, height, sliderPos,
347 minSliderPos, maxSliderPos, style, slider);
348 }

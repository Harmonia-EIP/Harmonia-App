41 void drawRotarySlider (juce::Graphics& g, int x, int y, int width, int height,
42 float sliderPos, float rotaryStartAngle, float rotaryEndAngle,
43 juce::Slider& slider) override
44 {
45 juce::ignoreUnused (slider);
46
47 auto radius = (float) juce::jmin (width / 2, height / 2) - 4.0f;
48 auto centreX = (float) x + (float) width * 0.5f;
49 auto centreY = (float) y + (float) height * 0.5f;
50 auto rx = centreX - radius;
51 auto ry = centreY - radius;
52 auto rw = radius * 2.0f;
53 auto angle = rotaryStartAngle + sliderPos * (rotaryEndAngle - rotaryStartAngle);
54
55 auto knobBg = findColour (AppColourIds::knobBgId);
56 auto knobOutline = findColour (AppColourIds::knobOutlineId);
57 auto knobPointer = findColour (AppColourIds::knobPointerId);
58
59 g.setColour (knobBg);
60 g.fillEllipse (rx, ry, rw, rw);
61
62 g.setColour (knobOutline);
63 g.drawEllipse (rx, ry, rw, rw, 1.5f);
64
65 juce::Path p;
66 auto pointerLength = radius * 0.7f;
67 auto pointerThickness = 3.0f;
68 p.addRectangle (-pointerThickness * 0.5f, -radius, pointerThickness, pointerLength);
69 p.applyTransform (juce::AffineTransform::rotation (angle).translated (centreX, centreY));
70
71 g.setColour (knobPointer);
72 g.fillPath (p);
73 }

43 const float right = area.getRight();
44 const float midY = area.getCentreY();
45 const float halfH = area.getHeight() * 0.40f;
46 const float w_ = right - left;
47
48 juce::Path p;

51 case Wave::Sine:
52 {
53 const int steps = 48;
54 for (int i = 0; i <= steps; ++i)
55 {
56 const float t = (float) i / (float) steps;
57 const float x = left + w_ * t;
58 const float y = midY - std::sin (t * juce::MathConstants<float>::twoPi) * halfH;

61 }
62 break;
63 }
64
65 case Wave::Triangle:

67 p.startNewSubPath (left, midY);

266 inline juce::NormalisableRange<float> makeLogRange (float minV,
267 float maxV,
268 float midpoint)
269 {
270 const float skew =
271 std::log (0.5f) /
272 std::log ((midpoint - minV) / (maxV - minV));
273
274 return { minV, maxV, 0.0001f, skew };

299 inline juce::AudioProcessorValueTreeState::ParameterLayout createLayout()
300 {
301 using APF = juce::AudioParameterFloat;
302 using APC = juce::AudioParameterChoice;
303 using Range = juce::NormalisableRange<float>;
304 using ParamID = juce::ParameterID;
305
306 constexpr int versionHint = 1;
307
308 juce::AudioProcessorValueTreeState::ParameterLayout layout;
309
310 const juce::StringArray waveforms { "Sine", "Triangle", "Saw", "Square" };
311 const juce::StringArray filterTypes { "Lowpass", "Bandpass", "Highpass" };
312
313 // Part1 Osc
314 layout.add (std::make_unique<APC> (ParamID { IDs::osc1Waveform, versionHint },
315 "Osc 1 Waveform", waveforms, 2)); // Saw
316 layout.add (std::make_unique<APC> (ParamID { IDs::osc2Waveform, versionHint },
317 "Osc 2 Waveform", waveforms, 1)); // Triangle
318 layout.add (std::make_unique<APF> (ParamID { IDs::oscMix, versionHint },
319 "Osc Mix",
320 Range { 0.0f, 1.0f, 0.0001f }, 0.5f));
321 layout.add (std::make_unique<APF> (ParamID { IDs::osc2Detune, versionHint },
322 "Osc 2 Detune",
323 Range { 0.0f, 100.0f, 0.01f }, 0.0f,
324 juce::AudioParameterFloatAttributes().withLabel ("cents")));
325 layout.add (std::make_unique<APF> (ParamID { IDs::noiseLevel, versionHint },
326 "Noise Level",
327 Range { 0.0f, 1.0f, 0.0001f }, 0.0f));
328
329 // Part 2 Filters
330 layout.add (std::make_unique<APF> (ParamID { IDs::filterCutoff, versionHint },
331 "Filter Cutoff",
332 makeLogRange (20.0f, 20000.0f, 1000.0f), 8000.0f,
333 juce::AudioParameterFloatAttributes().withLabel ("Hz")));
334 layout.add (std::make_unique<APF> (ParamID { IDs::filterResonance, versionHint },
335 "Filter Resonance",
336 Range { 0.0f, 0.95f, 0.0001f }, 0.1f));
337 layout.add (std::make_unique<APC> (ParamID { IDs::filterType, versionHint },
338 "Filter Type", filterTypes, 0));
339
340 // Part 3 Enveloppess
341 layout.add (std::make_unique<APF> (ParamID { IDs::ampAttack, versionHint },
342 "Amp Attack",
343 makeLogRange (1.0f, 5000.0f, 100.0f), 5.0f,
344 juce::AudioParameterFloatAttributes().withLabel ("ms")));
345 layout.add (std::make_unique<APF> (ParamID { IDs::ampDecay, versionHint },
346 "Amp Decay",
347 makeLogRange (10.0f, 5000.0f, 300.0f), 200.0f,
348 juce::AudioParameterFloatAttributes().withLabel ("ms")));
349 layout.add (std::make_unique<APF> (ParamID { IDs::ampSustain, versionHint },
350 "Amp Sustain",
351 Range { 0.0f, 1.0f, 0.0001f }, 0.8f));
352 layout.add (std::make_unique<APF> (ParamID { IDs::ampRelease, versionHint },
353 "Amp Release",
354 makeLogRange (10.0f, 10000.0f, 500.0f), 400.0f,
355 juce::AudioParameterFloatAttributes().withLabel ("ms")));
356 layout.add (std::make_unique<APF> (ParamID { IDs::filterEnvAmount, versionHint },
357 "Filter Env Amount",
358 Range { -1.0f, 1.0f, 0.0001f }, 0.0f));
359 layout.add (std::make_unique<APF> (ParamID { IDs::filterEnvDecay, versionHint },
360 "Filter Env Decay",
361 makeLogRange (10.0f, 5000.0f, 300.0f), 200.0f,
362 juce::AudioParameterFloatAttributes().withLabel ("ms")));
363
364 // Part 4 Lfos + modulations
365 layout.add (std::make_unique<APF> (ParamID { IDs::lfoRate, versionHint },
366 "LFO Rate",
367 makeLogRange (0.1f, 20.0f, 2.0f), 4.0f,
368 juce::AudioParameterFloatAttributes().withLabel ("Hz")));
369 layout.add (std::make_unique<APF> (ParamID { IDs::lfoToPitch, versionHint },
370 "LFO -> Pitch",
371 Range { 0.0f, 1.0f, 0.0001f }, 0.0f));
372 layout.add (std::make_unique<APF> (ParamID { IDs::lfoToCutoff, versionHint },
373 "LFO -> Cutoff",
374 Range { 0.0f, 1.0f, 0.0001f }, 0.0f));
375 layout.add (std::make_unique<APF> (ParamID { IDs::velocityToFilter, versionHint },
376 "Velocity -> Filter",
377 Range { 0.0f, 1.0f, 0.0001f }, 0.0f));
378 layout.add (std::make_unique<APF> (ParamID { IDs::distortionMix, versionHint },
379 "Distortion",
380 Range { 0.0f, 1.0f, 0.0001f, 1.8f }, 0.0f)); // skew>1 = expo
381 layout.add (std::make_unique<APF> (ParamID { IDs::reverbMix, versionHint },
382 "Reverb Mix",
383 Range { 0.0f, 1.0f, 0.0001f }, 0.2f));
384
385 return layout;
386 }
